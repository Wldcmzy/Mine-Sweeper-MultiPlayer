create table invitation( 
userID varchar(45) primary Key,  
invitationCode varchar(45),  
ifUsed  INT 
);

create table userInfo(
userID varchar(45) PRIMARY KEY, 
username varchar(45) UNIQUE, 
passwd varchar(45), 
ifOnline INT,  
clearCount INT, 
boomCount INT, 
FOREIGN KEY (userID) REFERENCES invitation (userID) 
);
